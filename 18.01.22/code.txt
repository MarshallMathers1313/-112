#include <iostream>
using namespace std;

{
	const int size = 6;
	int array[size] = { 1, 2, 3, 4, 5, 6 };

	for (int i = 0; i < size; i++)

	}

	//foreach

#include <iostream>

int main()
{
	srand(time(NULL));
	
	const int size = 7;
	int array[size] = { 3, 5, 9, 10, 123, 545, 888 };
	
	//Input in array
	cout << "Array: ";
	for (int i = 0; i < size; i++)
	{
		array[i] = rand() % 20;
		cout << array[i] << " ";
	}
	//O(n^2)-> Bubble Sort, Select Sort, Insert Sort
	//O(n* log(n)) - Binary Tree
	//O(n) -> Quick Sort, Marge Sort
	//O(log(n)) -> Binary Search
	//O(1) -> swap(array[j], array[j+1];
	

	//3, 5, 9, 10, 123, 545, 888-> Linear Search 7/2 = 3


	for (int i = 0; i < size; i++)
	{
		for (int j = 0; j < size - i - 1; j++)
		{
			if (array[j] > array[j+1])
			{
				/*int temp = array[j];
				array[j] = array[j + 1];
				array[j + 1] = temp;*/
				swap(array[j], array[j + 1]);
			}
		}
	}

	//foreach
	cout << "\nSorted Array: ";
	for (int i : array)
	{
		cout << i << " ";
	}

	return 0;
}

#include <iostream>
using namespace std;
int main()
{
	srand(time(NULL));
	
	const int size = 7;
	int array[size];
	
	//Input in array
	cout << "Array: ";
	for (int i = 0; i < size; i++)
	{
		array[i] = rand() % 20;
		cout << array[i] << " ";
	}

	//Select Sort
	for (int i = 0; i < size; i++)
	{
		int min = i;
		for (int j = i + 1; j < size; j++)
		{
			if (array[min] > array[j])//13 > 3
			{
				min = j;
			}
		}
		swap(array[i], array[min]);
	}

	//foreach
	cout << "\nSorted Array: ";
	for (int i : array)
	{
		cout << i << " ";
	}

	return 0;
}

	{
		array[i] = rand() % 20;
		cout << array[i] << " ";
	}

	//Insert Sort
	for (int i = 0; i < size; i++)
	{
		for (int j = i + 1; j < size; j++)
		{
			if (array[i] > array[j])
			{
				swap(array[i], array[j]);
			}
		}
	}

	//foreach
	cout << "\nSorted Array: ";
	for (int i : array)
	{
		cout << i << " ";
	}
	
}

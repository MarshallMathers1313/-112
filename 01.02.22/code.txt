#include <iostream>
using namespace std;

//Шаблоны функции
template <typename T1, typename T2>
T1 sum(T1 a, T2 b)
{
	return a + b;
}

template <typename T1, typename T2>
T1 _max(T1 a, T2 b)
{
	return a > b ? a : b;//A(65) < b(98)
}

int main()
{
	cout << sum(1, 2.5) << endl;
	cout << sum(2.2, 3.3) << endl;

	string str1 = "A";
	string str2 = "b";
	cout << sum(str1, str2) << endl;

	cout << _max(str1, str2) << endl;

	return 0;
}

#include <iostream>
using namespace std;

void menu()
{
	cout << "1 - Descending\n";
	cout << "0 - Ascending\n";
	cout << "Input your choice: ";
}

template <typename T>
void input(T array[], const int Size)
{
	for (int i = 0; i < Size; i++)
	{
		//cin >> array[i];
		array[i] = rand() % 20;
	}
}

template <typename T>
void output(T array[], const int Size)
{
	cout << "Output array: ";
	for (int i = 0; i < Size; i++)
	{
		cout << array[i] << " ";
	}
	cout << endl;
}

template <typename T>
void _sort(T array[], const int Size, bool x)
{
	for (int i = 0; i < Size; i++)
	{
		for (int j = i + 1; j < Size; j++)
		{
			if (x)
			{
				if (array[i] < array[j])
					swap(array[i], array[j]);
			}
			else
			{
				if (array[i] > array[j])
					swap(array[i], array[j]);
			}
		}
	}
}

int main()
{
	const int Size = 10;
	int array[Size];

	//Заполнение массива
	input(array, Size);
	//Вывод массива
	output(array, Size);

	//Выводим меню
	menu();
	bool x;
	cin >> x;

	//Сортировка по выбору
	_sort(array, Size, x);
	//Вывод массива
	output(array, Size);
}
